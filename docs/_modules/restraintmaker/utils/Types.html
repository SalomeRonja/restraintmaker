    def _calc_distance(self, atomA, atomB) -> float:
        """
            Calculate the distance between two atoms and their 3D positions.

        Parameters
        ----------
        atomA : u.Atom
        atomB : u.Atom


        Returns
        -------
        float
            returns the distance
        """
        return np.sqrt((atomA.x - atomB.x) ** 2 + (atomA.y - atomB.y) ** 2 + (atomA.z - atomB.z) ** 2)
